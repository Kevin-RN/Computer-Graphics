386  static Eigen::Matrix4f createPerspectiveMatrix (float fy, float in_aspect_ratio, float in_near_plane,float in_far_plane)
387  {
388  Eigen::Matrix4f out = Eigen::Matrix4f::Zero();
389 
390  const float
391  y_scale = (float)1.0/tan((fy/2.0)*(M_PI/180.0)),
392  x_scale = y_scale / in_aspect_ratio,
393  frustum_length = in_far_plane - in_near_plane;
394 
395  out(0,0) = x_scale;
396  out(1,1) = y_scale;
397  out(2,2) = -((in_far_plane + in_near_plane) / frustum_length);
398  out(3,2) = -1.0;
399  out(2,3) = -((2 * in_near_plane * in_far_plane) / frustum_length);
400 
401  return out;
