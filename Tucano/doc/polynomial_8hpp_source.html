448  void fitToPlane ( void )
449  {
450  if (key_positions.empty())
451  return;
452 
453  // compute center of mass
454  Eigen::Vector3f centroid = Eigen::Vector3f::Zero();
455  for (auto pt : key_positions)
456  {
457  centroid += pt.head(3);
458  }
459  centroid /= key_positions.size();
460 
461  Eigen::MatrixXf Y (3, key_positions.size());
462 
463  // fill scatter matrix
464  for (unsigned int i = 0; i < key_positions.size(); ++i)
465  {
466  Y.col(i) = key_positions[i].head(3) - centroid;
467  }
468 
469  Eigen::Matrix3f YYt = Y * Y.transpose();
470 
471  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3f> eigensolver(YYt);
472  Eigen::Vector3f normal = eigensolver.eigenvectors().col(0);
473  Eigen::Vector3f principal = eigensolver.eigenvectors().col(2);
474  normal.normalize();
475  principal.normalize();
476 
477  // compute rotation using normal, so it aligns with unit z
478  Eigen::Quaternionf rotZ = Eigen::Quaternionf::FromTwoVectors ( normal, Eigen::Vector3f::UnitZ() );
479 
480  // align largest eigenvector with unit x
481  Eigen::Quaternionf rotX = Eigen::Quaternionf::FromTwoVectors ( rotZ * principal, Eigen::Vector3f::UnitX() );
482 
483  // the above operation could probably be achieved just using the eigenvectors matrix as a change of basis matrix, should check for the sake of math (and fun)
484 
485  from3dTo2d = Eigen::Affine3f::Identity();
486  from3dTo2d.rotate (rotX);
487  from3dTo2d.rotate (rotZ);
488  from3dTo2d.translate ( -centroid );
489 
490  from2dTo3d = from3dTo2d.inverse();
491 
492  plane.resetModelMatrix();
493  plane.modelMatrix()->translate ( centroid );
494  plane.modelMatrix()->rotate (rotZ.inverse());
495  plane.modelMatrix()->rotate (rotX.inverse());
496 
497  plane_canonical.resetModelMatrix();
498  }

505  void computePolynomialRegression ( void )
506  {
507 
508  // prepare the matrices for the linear system
509  Eigen::MatrixXf A = Eigen::MatrixXf::Zero(polynomial_degree+1, polynomial_degree+1);
510  Eigen::VectorXf b = Eigen::VectorXf::Zero(polynomial_degree+1);
511 
512  Eigen::Vector3f plane_point;
513  for (unsigned int k = 0; k < key_positions.size(); ++k)
514  {
515  // project points to plane z = 0
516  plane_point = from3dTo2d * Eigen::Vector3f(key_positions[k].head(3));
517 
518  for (unsigned int j = 0; j <= polynomial_degree; j++) // row
519  {
520  for (unsigned int i = 0; i <= polynomial_degree; i++) // column
521  {
522  A(j,i) += pow ( plane_point[0], (polynomial_degree-j)+(polynomial_degree-i) );
523  }
524  b[j] += pow ( plane_point[0], polynomial_degree-j ) * plane_point[1];
525  }
526  }
527 
528  polynomial_coefficients.resize (polynomial_degree+1);
529  polynomial_coefficients = (A.transpose() * A).inverse() * A.transpose() * b;
530 
531  }
