450  Eigen::Quaternionf squad (int seg, float t)
451  {
452  Eigen::Quaternionf s0 = control_quaternions[seg];
453  Eigen::Quaternionf s1 = control_quaternions[seg+1];
454  Eigen::Quaternionf q0 = key_quaternions[seg];
455  Eigen::Quaternionf q1 = key_quaternions[seg+1];
456  Eigen::Quaternionf p0, p1, res;
457  float dot = 0.0;
458 
459  // before each slerp check if we are going the right direction (shortest path)
460  // otherwise multiply one of the quaternions by -1
461  // Note: this might not be necessary any more, since we are checking during log method
462 
463  dot = q0.w()*q1.w()+q0.vec().dot(q1.vec());
464  if (dot < 0)
465  {
466  q1.coeffs() *= -1.0;
467  }
468  p0 = q0.slerp(t, q1);
469 
470  dot = s0.w()*s1.w()+s0.vec().dot(s1.vec());
471  if (dot < 0)
472  {
473  s1.coeffs() *= -1.0;
474  }
475  p1 = s0.slerp(t, s1);
476 
477  p0.normalize();
478  p1.normalize();
479 
480  dot = p0.w()*p1.w()+p0.vec().dot(p1.vec());
481  if (dot < 0)
482  {
483  p1.coeffs() *= -1.0;
484  }
485  res = p0.slerp(2.0*t*(1.0-t), p1);
486  res.normalize();
487 
488  return res;

570  float arcLengthToTime (float s)
571  {
572  if (key_positions.size() < 2)
573  return -1.0;
574 
575  if (s < 0.0)
576  return 0;
577 
578  if (s > path_length)
579  return path_length;
580 
581  float arc_length = s;
582 
583  // find in which Bezir segment we should look
584  unsigned int segment = 0;
585  for (; segment < key_positions.size()-1; ++segment)
586  {
587  if (arc_lengths[segment+1][0] > arc_length)
588  break;
589  }
590 
591  // look inside Bezir segment in which linear approximation we should look for
592  unsigned int sub_seg = 0;
593  for (; sub_seg < arc_lengths[segment].size()-1; ++sub_seg)
594  {
595  if (arc_lengths[segment][sub_seg+1] > arc_length)
596  break;
597  }
598 
599  // convert from arc length (s) to time parameter (t)
600  // percentage inside sub segment
601  float alpha = (arc_length - arc_lengths[segment][sub_seg]) /
602  (arc_lengths[segment][sub_seg+1] - arc_lengths[segment][sub_seg]);
603 
604  // find parameter [0,1] inside Bezir segment
605  float t_ini = sub_seg / (float)(arc_lengths[segment].size()-1);
606  float t_end = (sub_seg+1) / (float)(arc_lengths[segment].size()-1);
607  float t_s = t_ini + alpha * (t_end - t_ini);
608 
609  // convert to global parameter of the path
610  t_s = t_s / (float)(key_positions.size()-1);
611 
612  // add time to beginning of segment
613  t_s += segment / (float)(key_positions.size()-1);
614 
615  return t_s;
616  }

783  void computeInnerControlPoints ( void )
784  {
785 
786  control_points_1.clear();
787  control_points_2.clear();
788 
789  // if there are only two keypoints it is a linear segment
790  // just distribute control points equally along segment
791  if (key_positions.size() == 2)
792  {
793  Eigen::Vector4f pt = 0.75*key_positions[0] + 0.25*key_positions[1];
794  control_points_1.push_back(pt);
795  control_points_1.push_back(pt);
796  pt << 0.25*key_positions[0] + 0.75*key_positions[1];
797  control_points_2.push_back(pt);
798  control_points_2.push_back(pt);
799  return;
800  }
801 
802  const int n = key_positions.size()-1;
803 
804  control_points_1.resize(n+1);
805  control_points_2.resize(n+1);
806 
807  // in theory this should be a nxn matrix, but since it is a tridiagonal matrix
808  // we are just storing the non-zero elements as a, b, and c
809  vector <float> a_orig;
810  vector <float> b_orig;
811  vector <float> c_orig;
812  vector <float> a;
813  vector <float> b;
814  vector <float> c;
815  vector <float> d;
816  a_orig.resize(n);
817  b_orig.resize(n);
818  c_orig.resize(n);
819  a.resize(n);
820  b.resize(n);
821  c.resize(n);
822  d.resize(n);
823 
824  // build the weight matrix, it is the same for all lines except first and last
825  a_orig[0] = 0.0;
826  b_orig[0] = 2.0;
827  c_orig[0] = 1.0;
828 
829  for (int i = 1; i < n-1; i++)
830  {
831  a_orig[i] = 1.0;
832  b_orig[i] = 4.0;
833  c_orig[i] = 1.0;
834  }
835 
836  a_orig[n-1] = 2.0;
837  b_orig[n-1] = 7.0;
838  c_orig[n-1] = 0.0;
839 
840  // solve using Thomas Algorithm for tridiagonal matrices
841  // solve the system three times, one for each coordinate (x, y, and z)
842  // we copy the coefficients each time to avoid recomputing them
843  for (int coord = 0; coord < 3; ++coord)
844  {
845  // build solution vector of linear system
846  d[0] = key_positions[0][coord] + 2.0 * key_positions[1][coord];
847  for (int i = 1; i < n-1; i++)
848  d[i] = 4.0*key_positions[i][coord] + 2.0*key_positions[i+1][coord];
849  d[n-1] = 8.0*key_positions[n-1][coord] + key_positions[n][coord];
850 
851  // copy coefficients, we are going to reuse them for every coordinate
852  a = a_orig;
853  b = b_orig;
854  c = c_orig;
855 
856  // forward sweep
857  c[0] = c[0]/b[0];
858  d[0] = d[0]/b[0];
859 
860  for (int i = 1; i < n; ++i)
861  {
862  float m = b[i] - a[i]*c[i-1];
863  c[i] = c[i] / m;
864  d[i] = (d[i] - a[i]*d[i-1]) / m;
865  }
866 
867  // back substitution
868  control_points_1[n-1][coord] = d[n-1];
869  for (int i = n-2; i >= 0; --i)
870  {
871  control_points_1[i][coord] = d[i] - c[i]*control_points_1[i+1][coord];
872  }
873  // add one last control point to facilitate future operations (same size as key vector)
874  control_points_1[n][coord] = control_points_1[n-1][coord];
875  }
876 
877  // set w = 1 for all control points
878  for (int i = 0; i <= n; ++i)
879  {
880  control_points_1[i][3] = 1.0;
881  }
882 
883  // control point 2 can be determined directly now
884  Eigen::Vector4f pt;
885  for (int i = 0; i < n-1; ++i)
886  {
887  pt = 2.0* key_positions[i+1] - control_points_1[i+1];
888  control_points_2[i] = pt;
889  }
890  pt = (key_positions[n] + control_points_1[n-1])*0.5;
891  control_points_2[n-1] = pt;
892  control_points_2[n] = pt;
893  }

903  void computeArcLength (void)
904  {
905  float divs = 100.0;
906  Eigen::Vector4f p0;
907  Eigen::Vector4f p1;
908  float dist = 0.0;
909  path_length = 0.0;
910 
911  arc_lengths.clear();
912 
913  p0 = key_positions[0];
914  for (unsigned int seg = 0; seg < key_positions.size()-1; ++seg)
915  {
916  vector <float> seg_lengths;
917 
918  // this is a pause position, just fill with constant length
919  if (key_intervals[seg] > 0.0)
920  {
921  float l = key_intervals[seg] / divs;
922  // already include extra segment since they are all the same
923  for (int i = 0; i < divs+1; ++i)
924  {
925  dist += l;
926  seg_lengths.push_back(dist);
927  }
928  p1 = key_positions[seg+1];
929  }
930  // otherwise divide the curve segment into divs linear segments
931  else
932  {
933  for (int i = 0; i < divs; ++i)
934  {
935  p1 = pointOnSegment( (float)i/divs, seg);
936  dist += (p1-p0).norm();
937 
938  seg_lengths.push_back(dist);
939  p0 = p1;
940  }
941  // to make life easier when converting from t to s,
942  // we also repeat the last sub segment length at the end
943  // even though it will be covered by the first sub segment
944  // of the next Bezir segment
