383  float arcLengthToTime (float s)
384  {
385  if (key_positions.size() < 2)
386  return -1.0;
387 
388  if (s < 0.0)
389  return 0;
390 
391  if (s > path_length)
392  return path_length;
393 
394  float arc_length = s;
395 
396  // find in which Beziér segment we should look
397  unsigned int segment = 0;
398  for (; segment < key_positions.size()-1; ++segment)
399  {
400  if (arc_lengths[segment+1][0] > arc_length)
401  break;
402  }
403 
404  // look inside Beziér segment in which linear approximation we should look for
405  unsigned int sub_seg = 0;
406  for (; sub_seg < arc_lengths[segment].size()-1; ++sub_seg)
407  {
408  if (arc_lengths[segment][sub_seg+1] > arc_length)
409  break;
410  }
411 
412  // convert from arc length (s) to time parameter (t)
413  // percentage inside sub segment
414  float alpha = (arc_length - arc_lengths[segment][sub_seg]) /
415  (arc_lengths[segment][sub_seg+1] - arc_lengths[segment][sub_seg]);
416 
417  // find parameter [0,1] inside Beziér segment
418  float t_ini = sub_seg / (float)(arc_lengths[segment].size()-1);
419  float t_end = (sub_seg+1) / (float)(arc_lengths[segment].size()-1);
420  float t_s = t_ini + alpha * (t_end - t_ini);
421 
422  // convert to global parameter of the path
423  t_s = t_s / (float)(key_positions.size()-1);
424 
425  // add time to beginning of segment
426  t_s += segment / (float)(key_positions.size()-1);
427 
428  return t_s;

558  void computeInnerControlPoints ( void )
559  {
560 
561  control_points_1.clear();
562  control_points_2.clear();
563 
564  // if there are only two keypoints it is a linear segment
565  // just distribute control points equally along segment
566  if (key_positions.size() == 2)
567  {
568  Eigen::Vector4f pt = 0.75*key_positions[0] + 0.25*key_positions[1];
569  control_points_1.push_back(pt);
570  control_points_1.push_back(pt);
571  pt << 0.25*key_positions[0] + 0.75*key_positions[1];
572  control_points_2.push_back(pt);
573  control_points_2.push_back(pt);
574  return;
575  }
576 
577  const int n = key_positions.size()-1;
578 
579  control_points_1.resize(n+1);
580  control_points_2.resize(n+1);
581 
582  // in theory this should be a nxn matrix, but since it is a tridiagonal matrix
583  // we are just storing the non-zero elements as a, b, and c
584  vector <float> a_orig;
585  vector <float> b_orig;
586  vector <float> c_orig;
587  vector <float> a;
588  vector <float> b;
589  vector <float> c;
590  vector <float> d;
591  a_orig.resize(n);
592  b_orig.resize(n);
593  c_orig.resize(n);
594  a.resize(n);
595  b.resize(n);
596  c.resize(n);
597  d.resize(n);
598 
599  // build the weight matrix, it is the same for all lines except first and last
600  a_orig[0] = 0.0;
601  b_orig[0] = 2.0;
602  c_orig[0] = 1.0;
603 
604  for (int i = 1; i < n-1; i++)
605  {
606  a_orig[i] = 1.0;
607  b_orig[i] = 4.0;
608  c_orig[i] = 1.0;
609  }
610 
611  a_orig[n-1] = 2.0;
612  b_orig[n-1] = 7.0;
613  c_orig[n-1] = 0.0;
614 
615  // solve using Thomas Algorithm for tridiagonal matrices
616  // solve the system three times, one for each coordinate (x, y, and z)
617  // we copy the coefficients each time to avoid recomputing them
618  for (int coord = 0; coord < 3; ++coord)
619  {
620  // build solution vector of linear system
621  d[0] = key_positions[0][coord] + 2.0 * key_positions[1][coord];
622  for (int i = 1; i < n-1; i++)
623  d[i] = 4.0*key_positions[i][coord] + 2.0*key_positions[i+1][coord];
624  d[n-1] = 8.0*key_positions[n-1][coord] + key_positions[n][coord];
625 
626  // copy coefficients, we are going to reuse them for every coordinate
627  a = a_orig;
628  b = b_orig;
629  c = c_orig;
630 
631  // forward sweep
632  c[0] = c[0]/b[0];
633  d[0] = d[0]/b[0];
634 
635  for (int i = 1; i < n; ++i)
636  {
637  float m = b[i] - a[i]*c[i-1];
638  c[i] = c[i] / m;
639  d[i] = (d[i] - a[i]*d[i-1]) / m;
640  }
641 
642  // back substitution
643  control_points_1[n-1][coord] = d[n-1];
644  for (int i = n-2; i >= 0; --i)
645  {
646  control_points_1[i][coord] = d[i] - c[i]*control_points_1[i+1][coord];
647  }
648  // add one last control point to facilitate future operations (same size as key vector)
649  control_points_1[n][coord] = control_points_1[n-1][coord];
650  }
651 
652  // set w = 1 for all control points
653  for (int i = 0; i <= n; ++i)
654  {
655  control_points_1[i][3] = 1.0;
656  }
657 
658  // control point 2 can be determined directly now
659  Eigen::Vector4f pt;
660  for (int i = 0; i < n-1; ++i)
661  {
662  pt = 2.0* key_positions[i+1] - control_points_1[i+1];
663  control_points_2[i] = pt;
664  }
665  pt = (key_positions[n] + control_points_1[n-1])*0.5;
666  control_points_2[n-1] = pt;
667  control_points_2[n] = pt;
668  }

678  void computeArcLength (void)
679  {
680  float divs = 100.0;
681  Eigen::Vector4f p0;
682  Eigen::Vector4f p1;
683  float dist = 0.0;
684  path_length = 0.0;
685 
686  arc_lengths.clear();
687 
688  p0 = key_positions[0];
689  for (unsigned int seg = 0; seg < key_positions.size()-1; ++seg)
690  {
691  vector <float> seg_lengths;
692 
693  // divide the curve segment into divs linear segments
694  for (int i = 0; i < divs; ++i)
695  {
696  p1 = pointOnSegment( (float)i/divs, seg);
697  dist += (p1-p0).norm();
698 
699  seg_lengths.push_back(dist);
700  p0 = p1;
701  }
702  // to make life easier when converting from t to s,
703  // we also repeat the last sub segment length at the end
704  // even though it will be covered by the first sub segment
705  // of the next Beziér segment
706  // note that we do not increment dist, since it will be done
707  // in the next iteration
708  p1 = key_positions[seg+1];
709  seg_lengths.push_back( dist + (p1-p0).norm() );
710  arc_lengths.push_back (seg_lengths);
711  }
712  // the last segment connecting to the final key position
713  // there is no segment starting at this key position, so it
714  // must be treated apart
715  p1 = key_positions[key_positions.size()-1];
716  dist += (p1-p0).norm();
717 
718  path_length = dist;
719 
720  // insert a last vector with only the total distance
721  // makes life easier when converting from arc length to time parameter
722  vector <float> last_seg;
723  last_seg.push_back (dist);
724  arc_lengths.push_back(last_seg);
725 
726  }
