198  void createGeometry (int subs_xy, int subs_z)
199  {
200 
201  vector< Eigen::Vector4f > vert;
202  vector< Eigen::Vector3f > norm;
203  vector< GLuint > faces;
204 
205  float x, y, theta;
206 
207  // create vertices for rings
208  for (int j = 0; j <= subs_z + 1; ++j)
209  {
210  float z = j * (1.0 / (float)(subs_z+1));
211  for (int i = 0; i < subs_xy; ++i)
212  {
213  theta = 2.0*M_PI*i/(float)subs_xy;
214  x = sin(theta);
215  y = cos(theta);
216  vert.push_back(Eigen::Vector4f(x, y, z, 1.0));
217  norm.push_back(Eigen::Vector3f(x, y, 0.0));
218  }
219  }
220 
221  // create a face with every three vertices between two rings
222  // every vertice creates a tringle with two vertices from lower ring and top ring, and another with two vertices from top ring and one from lower
223  for (int j = 0; j <= subs_z; ++j)
224  {
225  for (int i = 0; i < subs_xy; ++i)
226  {
227  faces.push_back(i + j*subs_xy); // current vertex
228  faces.push_back(i + (j+1)*subs_xy); // same vertex on lower ring
229  faces.push_back(((i+1)%subs_xy) + (j+1)*subs_xy); // next vertex on lower ring
230  faces.push_back((i+subs_xy-1)%subs_xy + j*subs_xy); // previous vertex on current ring
231  faces.push_back(i + (j+1)*subs_xy); // same vertex on lower ring
232  faces.push_back(i + j*subs_xy); // current vertex
233  }
234  }
235 
236  if (with_cap)
237  {
238  // create top cap
239  vert.push_back(Eigen::Vector4f(0.0, 0.0, 1.0, 1.0));
240  norm.push_back(Eigen::Vector3f(0.0, 0.0, 1.0));
241  int center_index = vert.size()-1;
242  int offset = vert.size();
243  for (int i = 0; i < subs_xy; ++i)
244  {
245  theta = 2.0*M_PI*i/(float)subs_xy;
246  x = sin(theta);
247  y = cos(theta);
248  vert.push_back(Eigen::Vector4f(x, y, 1.0, 1.0));
249  norm.push_back(Eigen::Vector3f(0.0, 0.0, 1.0));
250  }
251 
252  for (int i = 0; i < subs_xy; ++i)
253  {
254  faces.push_back(i+offset);
255  faces.push_back((i+1)%(subs_xy) + offset);
256  faces.push_back(center_index);
257  }
258  // create bottom cap
259  vert.push_back(Eigen::Vector4f(0.0, 0.0, 0.0, 1.0));
260  norm.push_back(Eigen::Vector3f(0.0, 0.0, -1.0));
261  center_index = vert.size()-1;
262  offset = vert.size();
263  for (int i = 0; i < subs_xy; ++i)
264  {
265  theta = 2.0*M_PI*i/(float)subs_xy;
266  x = sin(theta);
267  y = cos(theta);
268  vert.push_back(Eigen::Vector4f(x, y, 0.0, 1.0));
269  norm.push_back(Eigen::Vector3f(0.0, 0.0, -1.0));
270  }
271 
272  for (int i = 0; i < subs_xy; ++i)
273  {
274  faces.push_back(i+offset);
275  faces.push_back(center_index);
276  faces.push_back((i+1)%(subs_xy) + offset);
277  }
